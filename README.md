# Install

```sh
npm install --save react-octicons
```

# Usage

```jsx
import React, {Component} from 'react'
import {BeakerIcon, ArrowSmallDownIcon} from 'react-octicons'

export default class MyComponent extents Component {
  render() {
    return (
      <div>
        <h1><BeakerIcon/> Test Lab</h1>
      </div>
    )
  }
}
```
